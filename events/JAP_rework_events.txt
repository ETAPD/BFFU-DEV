add_namespace = JAP_rework

# The February 26 Incident (Ni-Niroku Jiken) - 26 February 1936
# Fires ~Feb 26 1936. Completing it auto-completes the JAP_niniroku_jiken focus,
# which installs Hirota Koki and applies the "Remnants of the Coup" spirit.
country_event = {
	id = JAP_rework.1
	title = JAP_rework.1.t
	desc = JAP_rework.1.desc
	picture = GFX_report_event_japanese_parade

	is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = JAP_rework.1.a # The rebellion must be put down.
		ai_chance = { base = 1 }
		complete_national_focus = JAP_niniroku_jiken
	}
}

# Hirota's Cabinet Falls - 2 February 1937
# Fires on 1937.2.2 if Hirota's era is still running. Completing it auto-completes
# JAP_hirotaoutoftime, which in turn installs Senjuro Hayashi via JAP_hayashi_enters_office.
country_event = {
	id = JAP_rework.2
	title = JAP_rework.2.t
	desc = JAP_rework.2.desc
	picture = GFX_report_event_japanese_parade

	is_triggered_only = yes
	fire_only_once = yes

	trigger = {
		has_completed_focus = JAP_niniroku_jiken
		NOT = { has_completed_focus = JAP_hirotaoutoftime }
	}

	option = {
		name = JAP_rework.2.a # Hirota's time is up.
		ai_chance = { base = 1 }
		complete_national_focus = JAP_hirotaoutoftime
	}
}

# Education Programs Success - fires 180 days after JAP_kyoikukaigicouncilestablished.
# The temporary Education Reforms drive matures into a permanent research bonus.
country_event = {
	id = JAP_rework.100
	title = JAP_rework.100.t
	desc = JAP_rework.100.desc
	picture = GFX_report_event_japanese_parade

	is_triggered_only = yes
	fire_only_once = yes

	option = {
		name = JAP_rework.100.a # Our investment in education has paid off.
		ai_chance = { base = 1 }
		if = {
			limit = { has_idea = JAP_education_reforms }
			remove_ideas = JAP_education_reforms
		}
		add_ideas = JAP_education_programs_success
	}
}
