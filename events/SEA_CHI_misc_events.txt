	desc = SEA_PRC_misc_events.14.desc
	picture = GFX_report_event_military_planning

	is_triggered_only = yes

	option = {
		name = SEA_PRC_misc_events.14.a

		every_core_state = {
			limit = {
				is_core_of = MAN
				is_fully_controlled_by = FROM
				NOT = {
					is_core_of = FROM
				}
				if = {
					limit = {
						FROM = {
							has_government = ROOT
						}
					}
					is_owned_by = FROM
				}
				else = {
					NOT = { is_owned_by = FROM }
				}
			}
			ROOT = {
				transfer_state = PREV
			}
		}
		add_opinion_modifier = {
			target = FROM
			modifier = ENG_returned_our_territory
		}
	}
}
