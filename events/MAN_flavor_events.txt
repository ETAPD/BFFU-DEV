	is_triggered_only = yes
	picture = GFX_report_event_manchukou.32

	immediate = {
		hidden_effect = {
			clr_country_flag = MAN_banditry_category_flag
			remove_dynamic_modifier = { modifier = MAN_banditry }
			add_dynamic_modifier = { modifier = MAN_banditry days = 720 }
			set_variable = { MAN_strength_of_partisan_movement_var = -0.25 }
			every_owned_state = {
				limit = { has_dynamic_modifier = { modifier = MAN_banditry_activity } }
				remove_dynamic_modifier = { modifier = MAN_banditry_activity }
				add_dynamic_modifier = { modifier = MAN_partisan_war days = 720 }
			}
			start_civil_war = {
				ideology = communism
				size = 0.2
				capital = 717
				states = { 610 714 761 }
				keep_all_characters = yes
				create_country_leader = {
					name = MAN_peoples_antijapanese_council
					picture = "gfx/leaders/MAN/MAN_collective_portrait_of_commi.dds"
					expire = "1965.1.1"
					ideology = stalinism
				}
				set_cosmetic_tag = MAN_peasant_uprising
				division_template = {
					name = "Resistance Fighters"
					priority = 1
					is_locked = no
					regiments = {
						infantry = { x = 0 y = 0 }
						infantry = { x = 0 y = 1 }
						infantry = { x = 0 y = 2 }
						infantry = { x = 1 y = 0 }
						infantry = { x = 1 y = 1 }
					}
				}
				capital_scope = {
					create_unit = {
						division = "name = \"Resistance Fighters\" division_template = \"Resistance Fighters\" start_experience_factor = 0.05 start_equipment_factor = 0.95" owner = PREV count = 10
					}
				}
			}
		}
	}

	option = {
		name = manchukou.32.a
	}
}
