on_actions = {

	# Schedule the Hirota-era events at game start.
	on_startup = {
		effect = {
			if = {
				limit = { country_exists = JAP }
				JAP = {
					country_event = { id = JAP_rework.1 days = 56 }  # 26 Feb 1936 - February 26 Incident
					country_event = { id = JAP_rework.2 days = 398 } # 2 Feb 1937 - Hirota's cabinet falls
				}
			}
		}
	}

}
