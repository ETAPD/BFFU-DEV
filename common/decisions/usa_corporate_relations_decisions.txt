	# NEGOTIATE / CONTRACT DECISIONS (75 PP each)
	# Bonus tier = trust level at time of signing.
	# Max 2 contracts active simultaneously.
	# Each contract lasts 1 year, then expires.
	##################################################

	USA_negotiate_with_kaiser = {
		icon = GFX_corp_usa_kaiser

		allowed = { original_tag = USA }

		visible = {
			has_completed_focus = USA_reconcile_with_domestic_corporations
		}

		available = {
			check_variable = { var = USA_kaiser_trust value = 0.5 compare = greater_than_or_equals }
			NOT = { has_country_flag = USA_kaiser_contract_active }
			NOT = { check_variable = { var = USA_active_contracts value = 2 compare = greater_than_or_equals } }
		}

		cost = 75
		days_remove = 0

		ai_will_do = { factor = 1 }

		complete_effect = {
			if = {
				limit = { check_variable = { var = USA_kaiser_trust value = 2.5 compare = greater_than_or_equals } }
				add_ideas = USA_kaiser_contract_3
				set_variable = { var = USA_kaiser_contract_tier value = 3 }
				# 3 civs in random states
				random_owned_controlled_state = {
					limit = { is_core_of = ROOT }
					add_extra_state_shared_building_slots = 1
					add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
				}
				random_owned_controlled_state = {
					limit = { is_core_of = ROOT }
					add_extra_state_shared_building_slots = 1
					add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
				}
				random_owned_controlled_state = {
					limit = { is_core_of = ROOT }
					add_extra_state_shared_building_slots = 1
					add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
				}
				custom_effect_tooltip = USA_kaiser_contract_civ_3_tt
			}
			else_if = {
				limit = { check_variable = { var = USA_kaiser_trust value = 1.5 compare = greater_than_or_equals } }
				add_ideas = USA_kaiser_contract_2
				set_variable = { var = USA_kaiser_contract_tier value = 2 }
				# 2 civs in random states
				random_owned_controlled_state = {
					limit = { is_core_of = ROOT }
					add_extra_state_shared_building_slots = 1
					add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
				}
				random_owned_controlled_state = {
					limit = { is_core_of = ROOT }
					add_extra_state_shared_building_slots = 1
					add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
				}
				custom_effect_tooltip = USA_kaiser_contract_civ_2_tt
			}
			else = {
				add_ideas = USA_kaiser_contract_1
				set_variable = { var = USA_kaiser_contract_tier value = 1 }
				# 1 civ in random state
				random_owned_controlled_state = {
					limit = { is_core_of = ROOT }
					add_extra_state_shared_building_slots = 1
					add_building_construction = { type = industrial_complex level = 1 instant_build = yes }
				}
				custom_effect_tooltip = USA_kaiser_contract_civ_1_tt
			}
			add_to_variable = { var = USA_active_contracts value = 1 }
			set_country_flag = USA_kaiser_contract_active
			custom_effect_tooltip = USA_contract_lasts_365_tt
			country_event = { id = usa_corp.1 days = 365 }
		}
	}
