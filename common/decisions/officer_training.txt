	bold_attack_rekless = {

		icon = ger_military_buildup

		available = {
			has_idea = bold_attack_spirit
			NOT = { 
				has_country_flag = officer_training_flag 
				has_country_flag = officer_bold_attack_flag
			}
			any_army_leader = {
				has_trait = command_school
				NOT = {
					has_trait = trait_reckless
				}
			}
		}
		
		custom_cost_trigger = {
			command_power > 34
		}
		custom_cost_text = decision_cost_CP_35
		
		days_remove = 365
		
		days_re_enable = 60

		visible = {
			has_idea = bold_attack_spirit			
		}

		complete_effect = {
			set_country_flag = officer_training_flag
			hidden_effect = {
				add_command_power = -35
			}

			random_army_leader = {
				limit = {
					has_trait = command_school
					# skill < 4
					NOT = {
						has_trait = trait_reckless
					}
				}
				set_character_flag = genneral_training_flag
			}
	
			custom_effect_tooltip = off_bold_reckless_tt
		}

		remove_effect = {
			random_army_leader = {
				limit = {
					has_character_flag = genneral_training_flag
					has_trait = command_school
				}
		
				remove_unit_leader_trait = command_school
				add_unit_leader_trait = trait_reckless
				clr_character_flag = genneral_training_flag
			}
			clr_country_flag = officer_training_flag
			set_country_flag = officer_bold_attack_flag
		}
	}
