# Imperial Liaison Conference — Army/Navy interservice rivalry (HoO2 inspired).
# One liaison action may be taken per 120 days (shared JAP_liaison_cd cooldown).
# Each tilts the JAP_liaison_* spirit ladder toward the Army or Navy and grants a
# matching bonus. Unlocked by "Begin the Imperial Liaison Conferences".

@liaison_cost = 40

JAP_imperial_liaison_conference = {

	#########################  ARMY  #########################

	# Pure tilt toward the Army.
	JAP_liaison_favour_the_army = {
		icon = GFX_decision_generic_army
		cost = @liaison_cost
		available = {
			NOT = { has_country_flag = JAP_liaison_cd }
			NOT = { has_idea = JAP_liaison_army_dominant }
		}
		complete_effect = {
			JAP_liaison_towards_army = yes
			army_experience = 25
			set_country_flag = { flag = JAP_liaison_cd value = 1 days = 120 }
		}
		ai_will_do = { factor = 1 }
	}

	# Divert steel and slipway labour to the arsenals.
	JAP_liaison_steel_for_the_guns = {
		icon = GFX_decision_generic_army
		cost = @liaison_cost
		available = { NOT = { has_country_flag = JAP_liaison_cd } }
		complete_effect = {
			JAP_liaison_towards_army = yes
			random_owned_controlled_state = {
				limit = { is_core_of = JAP }
				add_extra_state_shared_building_slots = 2
				add_building_construction = { type = arms_factory level = 2 instant_build = yes }
			}
			set_country_flag = { flag = JAP_liaison_cd value = 1 days = 120 }
		}
		ai_will_do = { factor = 1 }

			add_manpower = 50000

		available = {
			NOT = { has_country_flag = JAP_liaison_cd }
			NOT = { has_idea = JAP_liaison_balanced }
		}
		complete_effect = {
			if = {
				limit = { OR = { has_idea = JAP_liaison_army_favoured  has_idea = JAP_liaison_army_dominant } }
