# Rapid industrialization at a price: every project costs political power
# and stability up front, and carries a hidden risk of industrial accidents
# (man_behemoth.31 minor / .32 major — the major one destroys a factory).
# Each completed project adds +20 plan progress; at 100 the plan concludes
# (man_behemoth.1 / .2) and the next stage unlocks.
# All economy effects accumulate in the MAN_industrial_behemoth spirit.

MAN_ayukawa_vs_matsuoka_category = {

	# Resolution mission — at timeout, whoever holds the bureaucracy wins
	MAN_bureaucratic_confrontation = {
		icon = GFX_decision_category_MAN_second_fyp
		priority = 100
		allowed = { original_tag = MAN }
		available = { hidden_trigger = { always = no } }
		activation = { has_country_flag = MAN_ayukawa_vs_matsuoka_flag }
		is_good = no
		fire_only_once = yes
		days_mission_timeout = 540
		cancel_trigger = { NOT = { has_country_flag = MAN_ayukawa_vs_matsuoka_flag } }
		timeout_effect = {
			if = {
				limit = { check_variable = { MAN_ayukawa_vs_matsuoka_var > 74 } }
				country_event = { id = manchukou.134 hours = 6 }
				country_event = { id = manchukou.127 days = 7 }
				JAP = { country_event = { id = manchukou.127 days = 7 } }
			}
			else_if = {
				limit = { check_variable = { MAN_ayukawa_vs_matsuoka_var > 49 } }
				country_event = { id = manchukou.127 hours = 6 }
				JAP = { country_event = { id = manchukou.127 hours = 6 } }
			}
			else_if = {
				limit = { check_variable = { MAN_ayukawa_vs_matsuoka_var > 25 } }
				country_event = { id = manchukou.132 hours = 6 }
				JAP = { country_event = { id = manchukou.132 hours = 6 } }
			}
			else = {
				country_event = { id = manchukou.136 hours = 6 }
				country_event = { id = manchukou.132 days = 7 }
				JAP = { country_event = { id = manchukou.132 days = 7 } }
			}
		}
	}

	MAN_support_ayukawa = {
