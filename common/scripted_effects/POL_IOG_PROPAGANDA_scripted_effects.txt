# Poland
	POL_IOG_pick_propaganda_campaign_slot_effect = {
		hidden_effect = {
			if = {
				limit = {
					check_variable = { POL_IOG_poster_left_icon_frame = POL_IOG_propaganda_empty_slot_index }
				}
				set_variable = { POL_IOG_poster_left_icon_frame = POL_IOG_new_propaganda_campaign }
				#Ageing posters
				set_variable = { POL_IOG_propaganda_left_poster_to_age = POL_IOG_poster_left_icon_frame } #This is used to prevent the system to break if a propaganda campaign is cancelled
			}
			else_if = {
				limit = {
					check_variable = { POL_IOG_poster_center_icon_frame = POL_IOG_propaganda_empty_slot_index }
				}
				set_variable = { POL_IOG_poster_center_icon_frame = POL_IOG_new_propaganda_campaign }
				#Ageing posters
				set_variable = { POL_IOG_propaganda_center_poster_to_age = POL_IOG_poster_center_icon_frame } #This is used to prevent the system to break if a propaganda campaign is cancelled
			}
			set_variable = { POL_IOG_new_propaganda_campaign = 0 } # Reset var
			add_to_variable = { POL_IOG_propaganda_campaigns_taken = 1 } #Update propaganda campaigns counter
		}
	}

	POL_IOG_reset_propaganda_campaign_slot_effect = {
		hidden_effect = {
			if = {
				limit = {
					check_variable = { POL_IOG_poster_left_icon_frame = POL_IOG_propaganda_campaign_to_remove }
				}
				set_variable = { POL_IOG_poster_left_icon_frame = POL_IOG_propaganda_empty_slot_index }
				#Clear aged posters
				clear_variable = POL_IOG_propaganda_left_poster_to_age
				clr_country_flag = POL_IOG_propaganda_left_slot_is_old_flag
			}
			else_if = {
				limit = {
					check_variable = { POL_IOG_poster_center_icon_frame = POL_IOG_propaganda_campaign_to_remove }
				}
				set_variable = { POL_IOG_poster_center_icon_frame = POL_IOG_propaganda_empty_slot_index }
				#Clear aged posters
				clear_variable = POL_IOG_propaganda_center_poster_to_age
				clr_country_flag = POL_IOG_propaganda_center_slot_is_old_flag
			}
			set_variable = { POL_IOG_propaganda_campaign_to_remove = 0 } # Reset var
		}
	}

	POL_IOG_unlock_propaganda_campaign_slot_effect = {
		hidden_effect = {
			if = {
				limit = {
					check_variable = { POL_IOG_poster_left_icon_frame = POL_IOG_propaganda_locked_slot_index }
				}
				set_variable = { POL_IOG_poster_left_icon_frame = POL_IOG_propaganda_empty_slot_index }
			}
			else_if = {
				limit = {
					check_variable = { POL_IOG_poster_center_icon_frame = POL_IOG_propaganda_locked_slot_index }
				}
				set_variable = { POL_IOG_poster_center_icon_frame = POL_IOG_propaganda_empty_slot_index }
			}
		}
	}
