# ── Third New Deal — single variable-driven spirit ─────────────────────────
# Base values are set on grant; acts/paths/events apply deltas to the
# USA_tnd_* variables and force a refresh so the changes display live.

# Grants the Third New Deal spirit and initialises its base values.
USA_grant_third_new_deal = {
	hidden_effect = {
		set_variable = { USA_tnd_stability = 0 }
		set_variable = { USA_tnd_infra     = 0 }
		set_variable = { USA_tnd_ic        = 0 }
		set_variable = { USA_tnd_ppcost    = 0 }
		set_variable = { USA_tnd_consumer  = 0 }
		set_variable = { USA_tnd_factory   = 0 }
	}
	custom_effect_tooltip = USA_add_third_new_deal_tt
	add_to_variable = { var = USA_tnd_stability value = 0.10 tooltip = stability_factor_tt }
	add_to_variable = { var = USA_tnd_infra     value = 0.50 tooltip = production_speed_infrastructure_factor_tt }
	add_to_variable = { var = USA_tnd_ic        value = 0.40 tooltip = production_speed_industrial_complex_factor_tt }
	add_to_variable = { var = USA_tnd_ppcost    value = 0.35 tooltip = political_power_cost_tt }
	hidden_effect = { add_dynamic_modifier = { modifier = USA_third_new_deal } }
	force_update_dynamic_modifier = yes
}

# Removes the Third New Deal spirit and clears its variables.
USA_remove_all_third_new_deal = {
	if = {
		limit = { has_dynamic_modifier = { modifier = USA_third_new_deal } }
		remove_dynamic_modifier = { modifier = USA_third_new_deal }
	}
	clear_variable = USA_tnd_stability
	clear_variable = USA_tnd_infra
	clear_variable = USA_tnd_ic
	clear_variable = USA_tnd_ppcost
	clear_variable = USA_tnd_consumer
	clear_variable = USA_tnd_factory
}
