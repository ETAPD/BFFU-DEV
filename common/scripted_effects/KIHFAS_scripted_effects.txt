KIH_nostrum_lvl_down = {
	if = {
		limit = { has_idea = ITA_mare_nostrum_max }
		swap_ideas = {
			remove_idea = ITA_mare_nostrum_max
			add_idea = ITA_mare_nostrum_minus_one
		}
	}
	else_if = {
		limit = { has_idea = ITA_mare_nostrum_minus_one }
		swap_ideas = {
			remove_idea = ITA_mare_nostrum_minus_one
			add_idea = ITA_mare_nostrum_minus_two
		}
	}
	else_if = {
		limit = { has_idea = ITA_mare_nostrum_minus_two }
		swap_ideas = {
			remove_idea = ITA_mare_nostrum_minus_two
			add_idea = ITA_mare_nostrum_minus_three
		}
	}
	else_if = {
		limit = { has_idea = ITA_mare_nostrum_minus_three }
		swap_ideas = {
			remove_idea = ITA_mare_nostrum_minus_three
			add_idea = ITA_mare_nostrum_minus_four
		}
	}
	else_if = {
		limit = { has_idea = ITA_mare_nostrum_minus_four }
		swap_ideas = {
			remove_idea = ITA_mare_nostrum_minus_four
			add_idea = ITA_mare_nostrum_minus_five
		}
	}
	else_if = {
		limit = { has_idea = ITA_mare_nostrum_minus_five }
		swap_ideas = {
			remove_idea = ITA_mare_nostrum_minus_five
			add_idea = ITA_mare_nostrum_minus_six
		}
	}
	else_if = {
		limit = { has_idea = ITA_mare_nostrum_minus_six }
		swap_ideas = {
			remove_idea = ITA_mare_nostrum_minus_six
			add_idea = ITA_mare_nostrum_minus_seven
		}
	}
}
