# Plan completion checks
MAN_fyp1_completion_check = {
	if = {
		limit = {
			check_variable = { MAN_fyp1_progress > 99 }
			NOT = { has_country_flag = MAN_first_fyp_complete }
		}
		set_country_flag = MAN_first_fyp_complete
		country_event = { id = man_behemoth.1 days = 3 }
	}
}

MAN_fyp2_completion_check = {
